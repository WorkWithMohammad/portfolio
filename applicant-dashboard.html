/* ============================================================ dashboard.js Shared dashboard helpers: notes panel, documents panel. Per-role rendering lives in the page files. ============================================================ */ (function (global) { function renderNotesPanel(container, userId) { function paint() { const notes = Store.getNotes(userId); container.innerHTML = `
${notes.length ? notes.map(n => `

${UI.escapeHtml(n.title)}

${UI.timeAgo(n.createdAt)}

${UI.escapeHtml(n.body)}

`).join('') : `
No notes yet. Add your first one above.
`}
`; container.querySelector('#noteForm').addEventListener('submit', e => { e.preventDefault(); const f = new FormData(e.target); Store.addNote(userId, { title: f.get('title').trim(), body: f.get('body').trim() }); UI.toast('Note added', 'success'); paint(); }); container.querySelectorAll('[data-del]').forEach(b => b.addEventListener('click', () => { Store.deleteNote(userId, b.dataset.del); paint(); })); } paint(); } function renderDocsPanel(container, userId, { label='Documents' } = {}) { function paint() { const docs = Store.getDocs(userId); container.innerHTML = `
${docs.length ? docs.map(d => `

${UI.escapeHtml(d.name)}

${UI.escapeHtml(d.type||'file')}${UI.bytes(d.size||0)}${UI.timeAgo(d.uploadedAt)}
Download
`).join('') : `
No ${label.toLowerCase()} uploaded yet.
`}
`; container.querySelector('#docForm input[type=file]').addEventListener('change', async (e) => { const file = e.target.files[0]; if (!file) return; if (file.size > 4 * 1024 * 1024) { UI.toast('File too large (max ~4MB).', 'error'); return; } try { const dataUrl = await UI.readFileAsDataURL(file); Store.addDoc(userId, { name: file.name, type: file.type, size: file.size, dataUrl }); UI.toast('File uploaded', 'success'); paint(); } catch { UI.toast('Could not read file.', 'error'); } }); container.querySelectorAll('[data-del]').forEach(b => b.addEventListener('click', () => { Store.deleteDoc(userId, b.dataset.del); paint(); })); } paint(); } global.Dashboard = { renderNotesPanel, renderDocsPanel }; })(window);
